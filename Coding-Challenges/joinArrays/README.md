# Join Arrays

Combine the three arrays into one with all headers present

```js
const arr1 = [
  ["name", "id", "age", "weight", "Cool"],
  ["Susan", "3", "20", "120", true],
  ["John", "1", "21", "150", true],
  ["Bob", "2", "23", "90", true],
  ["Ben", "4", "20", "100", true],
]

const arr2 = [
  ["name", "id", "height"],
  ["Bob", "2", "50"],
  ["John", "1", "45"],
  ["Ben", "4", "43"],
  ["Susan", "3", "48"],
]

const arr3 = [
  ["name", "id", "siblings"],
  ["Bob", "2", "yes"],
  ["John", "1", "yes"],
]
```
